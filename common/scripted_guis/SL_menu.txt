scripted_gui = {

	SL_scanlines_button_gui = {
		context_type = player_context
		parent_window_token = top_bar 
		window_name = "SL_scanlines_button_gui_container"

		ai_check = {
			always = no
		}

		triggers = {
			SL_button_activate_click_enabled = {
				NOT = { has_country_flag = scanlines_menu_open }
			}		
		}

		effects = {
			SL_button_activate_click = {
				set_country_flag = scanlines_menu_open
				if = {
					limit = {
						NOT = {
							OR = {
								has_country_flag = SL_no_lines_flag
								has_country_flag = SL_25_lines_flag
								has_country_flag = SL_50_lines_flag
								has_country_flag = SL_100_lines_flag
							}
						}
					}
					set_country_flag = SL_no_lines_flag
				}
			 }
		 }
	}
	
	SL_scanlines_effect_gui = {
		context_type = player_context
		window_name = "SL_scanlines_effect_gui_container_ms"

		ai_check = {
			always = no
		}

		triggers = {
			hoi4_scanlines_25_visible = {
				has_country_flag = SL_25_lines_flag
			}
			
			hoi4_scanlines_50_visible = {
				has_country_flag = SL_50_lines_flag
			}
			
			hoi4_scanlines_100_visible = {
				has_country_flag = SL_100_lines_flag
			}	
			hoi4_scanlines_extended_25_visible = {
				has_country_flag = SL_25_lines_flag
			}
			
			hoi4_scanlines_extended_50_visible = {
				has_country_flag = SL_50_lines_flag
			}
			
			hoi4_scanlines_extended_100_visible = {
				has_country_flag = SL_100_lines_flag
			}			
			hoi4_scanlines_extended_25_2_visible = {
				has_country_flag = SL_25_lines_flag
			}
			
			hoi4_scanlines_extended_50_2_visible = {
				has_country_flag = SL_50_lines_flag
			}
			
			hoi4_scanlines_extended_100_2_visible = {
				has_country_flag = SL_100_lines_flag
			}
		}
	}


	SL_scanlines_menu_gui = {
		context_type = player_context
		window_name = "SL_scanlines_menu_gui_container"

		visible = {
			has_country_flag = scanlines_menu_open
		}
		
		triggers = {
			button_scanlines_off_click_enabled = {
				NOT = { has_country_flag = SL_no_lines_flag }
			}
			
			button_25scanlines_click_enabled = {
				NOT = { has_country_flag = SL_25_lines_flag }
			}
			
			button_50scanlines_click_enabled = {
				NOT = { has_country_flag = SL_50_lines_flag }
			}
			
			button_100scanlines_click_enabled = {
				NOT = { has_country_flag = SL_100_lines_flag }
			}
		}		

		effects = {
			SL_button_deactivate_click = {
				clr_country_flag = scanlines_menu_open
			}
			
			button_scanlines_off_click = {
				clr_country_flag = SL_25_lines_flag
				clr_country_flag = SL_50_lines_flag
				clr_country_flag = SL_100_lines_flag
				
				set_country_flag = SL_no_lines_flag
			}	
			
			button_25scanlines_click = {
				clr_country_flag = SL_no_lines_flag
				clr_country_flag = SL_50_lines_flag
				clr_country_flag = SL_100_lines_flag
				
				set_country_flag = SL_25_lines_flag
			}

			button_50scanlines_click = {
				clr_country_flag = SL_no_lines_flag
				clr_country_flag = SL_25_lines_flag
				clr_country_flag = SL_75_lines_flag
				clr_country_flag = SL_100_lines_flag
				
				set_country_flag = SL_50_lines_flag
			}

			button_100scanlines_click = {
				clr_country_flag = SL_no_lines_flag
				clr_country_flag = SL_25_lines_flag
				clr_country_flag = SL_50_lines_flag
				
				set_country_flag = SL_100_lines_flag
			}
		}
	}
}
